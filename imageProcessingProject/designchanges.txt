save (@ImageName) (@LayerName) @FileName
load @ImageName

set-current-layer (@ImageName) @LayerName
add-layer (@ImageName) @LayerName (@LayerToCopy)
add-image-as-layer (@ImageName) @LayerName @FileName
move-layer (@ImageName) @LayerName @NewPosition
remove-layer (@ImageName) @LayerPosition
save-as-image (@ImageName) @BlendType @OutputType @FileName
